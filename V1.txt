// L298N control pins (match your current wiring)
const int IN1 = 2;   // L298N IN1
const int IN2 = 3;   // L298N IN2
const int IN3 = 4;   // L298N IN3
const int IN4 = 5;   // L298N IN4
const int ENA = 9;   // L298N ENA (PWM)
const int ENB = 10;  // L298N ENB (PWM)

// Buttons (hold-to-run)
const int BTN_FWD = 7;  // Forward button
const int BTN_REV = 8;  // Reverse button

const int SPEED = 150;  // Motor speed (0â€“255)

void setup() {
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  pinMode(ENA, OUTPUT);
  pinMode(ENB, OUTPUT);

  // Use internal pull-ups (pressed = LOW)
  pinMode(BTN_FWD, INPUT_PULLUP);
  pinMode(BTN_REV, INPUT_PULLUP);

  stopMotors();  // Ensure motors are off at startup
}

void loop() {
  bool forwardPressed = (digitalRead(BTN_FWD) == LOW);
  bool reversePressed = (digitalRead(BTN_REV) == LOW);

  if (forwardPressed && !reversePressed) {
    runForward(SPEED);
  }
  else if (reversePressed && !forwardPressed) {
    runReverse(SPEED);
  }
  else {
    stopMotors();
  }
}

void runForward(int spd) {
  analogWrite(ENA, spd);
  analogWrite(ENB, spd);

  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);

  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void runReverse(int spd) {
  analogWrite(ENA, spd);
  analogWrite(ENB, spd);

  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);

  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
}

void stopMotors() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);

  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
}
